461  * Change the behavior of a PropagatedPlanet.
462  */
463 void
464 PropagatedPlanet::set_mode (
465  void)
466 {
467 
468  // Switching from the body being the source to the planet is illegal.
469  if ((((mode == TransFromBody_RotFromPlanet) ||
470  (mode == TransFromBody_RotFromBody)) &&
471  ((commanded_mode == TransFromPlanet_RotFromPlanet) ||
472  (commanded_mode == TransFromPlanet_RotFromBody))) ||
473  (((mode == TransFromPlanet_RotFromBody) ||
474  (mode == TransFromBody_RotFromBody)) &&
475  ((commanded_mode == TransFromPlanet_RotFromPlanet) ||
476  (commanded_mode == TransFromBody_RotFromPlanet)))) {
477  MessageHandler::fail (
478  __FILE__, __LINE__, EphemeridesMessages::inconsistent_setup,
479  "Illegal attempt to switch PropagatedPlanet to ephemeris mode.");
480  return;
481  }
482 
483  // Enable/disable the ephemeris objects per the mode.
484  // Enabled means that the planet frame depends on the body frame;
485  // disabled means the body frame depends on the planet frame.
486  if ((commanded_mode == TransFromBody_RotFromPlanet) ||
487  (commanded_mode == TransFromBody_RotFromBody)) {
488  ephem_planet.enable ();
489  }
490  else {
491  ephem_planet.disable ();
492  }
493 
494  if ((commanded_mode == TransFromPlanet_RotFromBody) ||
495  (commanded_mode == TransFromBody_RotFromBody)) {
496  ephem_orient.enable ();
497  }
498  else {
499  ephem_orient.disable ();
500  }
501 
502  mode = commanded_mode;
503  dyn_manager->ephem_note_tree_status_change ();
504 }

552  * Update ephemerides for subscribed items.
553  */
554 void
555 PropagatedPlanet::ephem_update (
556  void)
557 {
558 
559  if (! active) {
560  return;
561  }
562 
563  // Initialize the dyn body if this hasn't been done yet.
564  if (! initialized) {
565  dyn_manager->initialize_dyn_body (body);
566  initialized = true;
567  }
568 
569  // Get the time of this update.
570  update_time = time_dyn->seconds;
571 
572  // Update the dependent translational and rotational states from the sources.
573  ephem_planet.update (update_time);
574  ephem_orient.update (update_time);
575 
576  // Not in full body mode => need to propagate state throughout the body.
577  if (mode != TransFromBody_RotFromBody) {
578  body.propagate_state ();
579  }
580 
581  // Finally, change the mode to the commanded mode if needed.
582  // This must be done *after* the two update calls above to accommodate
583  // a change from FromPlanet to FromBody at post-initialization time.
584  if (commanded_mode != mode) {
585  set_mode ();
586  }
