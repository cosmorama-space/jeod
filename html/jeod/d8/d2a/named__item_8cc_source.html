73 // va_construct_name: Construct a name as a dot-conjoined string.
74 char *
75 NamedItem::va_construct_name (
76  const char * name_item,
77  va_list args)
78 {
79  std::size_t num_items; // -- Number of arguments
80  bool terminated; // -- Arguments terminated with NULL?
81 
82  // Protect against a null first argument.
83  if (name_item == nullptr) {
84  va_end (args);
85  return nullptr;
86  }
87 
88  std::string name(name_item); // -- The constructed name
89 
90  // Initialize the argument stack.
91  num_items = 0;
92  terminated = false;
93 
94  // Extract items from the argument stack, ending at the null pointer
95  // or when too many arguments have been pulled off
96  // (The latter probably means the caller forgot to end with a null).
97  while (1) {
98  const char * item = va_arg (args, const char *);
99 
100  // NULL terminates the argument list.
101  if (item == nullptr) {
102  terminated = true;
103  break;
104  }
105 
106  // Construct the name via join ('.', @arguments)
107  name += ".";
108  name += item;
109 
110  // Abort if too many arguments were passed (usually non NULL at end).
111  if (num_items >= MAX_NAME_ITEMS) {
112  break;
113  }
114 
115  num_items++;
116  }
117 
118  // Close the argument stack. This is needed here rather in the calling
119  // function in case an error occurs. The call to MessageHandler::fail will
120  // be fouled up if the argument stack is not closed.
121  va_end (args);
122 
123 
124  // Die if the null pointer was not found.
125  if (! terminated) {
126  MessageHandler::fail (
127  __FILE__, __LINE__, NamedItemMessages::bad_args,
128  "Too many arguments to NamedItem::construct_name");
129  return nullptr;
130  }
131 
132 
133  // Successfully extracted num_items arguments.
134  char * new_name = JEOD_ALLOC_PRIM_ARRAY (name.length()+1, char);
135  snprintf(new_name, name.length()+1, "%s", name.c_str());
136 
137  return new_name;
138 }

141 // suffix: Find the part of the name that follows the prefix.
142 const char *
143 NamedItem::suffix (
144  const char * prefix,
145  const char * name)
146 {
147  std::string nprefix(prefix);
148  std::size_t prefix_len = nprefix.length();
149 
150  // Return the full name if
151  // - the name is NULL (sanity check) or
152  // - the name does not start with "prefix."
153  if ((name == nullptr) ||
154  (std::strncmp (name, prefix, prefix_len) != 0) ||
155  (name[prefix_len] != '.')) {
156  return name;
157  }
158 
159  // Name starts with "prefix.": return the suffix.
160  else {
161  return &name[prefix_len+1];
162  }
