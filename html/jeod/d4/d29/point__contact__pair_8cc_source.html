65 /**
66  * Determine if contact has occurred between the facets of the pair.
67  */
68 void
69 PointContactPair::in_contact(
70  void)
71 {
72  double subject_contact_point[3];
73  double target_contact_point[3];
74  double vec[3], nvec[3];
75  double target_mag;
76 
77  // find direction vector to target from relstate
78  Vector3::normalize(rel_state.rel_state.trans.position, vec);
79  // pass direction vector to the subject so it can calculate the closest point.
80  point_subject->calculate_contact_point(vec);
81 
82  // rotate direction vector to target frame
83  Vector3::transform(rel_state.rel_state.rot.T_parent_this,
84  rel_state.rel_state.trans.position, nvec);
85  Vector3::negate(nvec, nvec);
86  Vector3::normalize(nvec, vec);
87  point_target->calculate_contact_point(vec);
88 
89  // get the contact points calculated above
90  Vector3::copy(point_subject->contact_point, subject_contact_point);
91  Vector3::copy(point_target->contact_point, target_contact_point);
92 
93  // convert target point into subject frame through rotation vector addition
94  Vector3::transform_transpose (rel_state.rel_state.rot.T_parent_this,
95  target_contact_point, vec);
96 
97  Vector3::sum(rel_state.rel_state.trans.position, vec, target_contact_point);
98  // find the distance to the target_contact_point
99  target_mag = Vector3::vmag(target_contact_point);
100 
101  // compare length to target point with subject radius
102  if (point_subject->radius > target_mag) {
103  double rel_velocity[3];
104  // find the vector that characterizes the interpenetration
105  Vector3::diff(target_contact_point, subject_contact_point, vec);
106  // find the relative velocity of the target in subject frame
107  Vector3::cross (rel_state.rel_state.rot.ang_vel_this, subject_contact_point, rel_velocity);
108  Vector3::diff (rel_velocity, rel_state.rel_state.trans.velocity, rel_velocity);
109  // calculate the forces on the facets
110  interaction->calculate_forces(point_subject, point_target, &rel_state, vec, rel_velocity);
111 
112  }
113 
114  return;
115 }
