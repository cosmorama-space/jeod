108 /**
109  * Construct the skew symmetric cross product matrix:
110  * mat[i][k] = epsilon_ijk vec[j], epsilon_ijk is the Levi-Cevita symbol
111  * \param[in] vec Vector
112  * \param[out] cross_mat Cross product matrix
113  */
114 inline void
115 Matrix3x3::cross_matrix (
116  double const vec[3],
117  double cross_mat[3][3])
118 {
119  cross_mat[0][0] = cross_mat[1][1] = cross_mat[2][2] = 0.0;
120 
121  // ijk = 012, 210
122  cross_mat[0][2] = vec[1];
123  cross_mat[2][0] = -vec[1];
124 
125  // ijk = 120, 021
126  cross_mat[1][0] = vec[2];
127  cross_mat[0][1] = -vec[2];
128 
129  // ijk = 201, 102
130  cross_mat[2][1] = vec[0];
131  cross_mat[1][2] = -vec[0];
132 
133  return;
134 }
