107  * Updates the model at each time step.
108  * \param[in] veh_struc_frame The vehicle structural reference frame.
109  * \param[in] center_grav position of center of mass\n Units: M
110  * \param[in] scale_factor the scale between the simulator time and the dynamic time.

112  */
113 void
114 RadiationPressure::update (
115  RefFrame & veh_struc_frame,
116  double center_grav[3],
117  double scale_factor,
118  double real_time)
119 {
120  if (!active) {
121  return;
122  }
123 
124  // ThermalFacetRider cycle-time is the time needed for integration purposes,
125  // and includes "catchup" whenever the rate at which time advances changes.
126  ThermalFacetRider::cycle_time =
127  JeodSimulationInterface::get_job_cycle() * scale_factor;
128 
129 
130  source.calculate_flux (veh_struc_frame,
131  center_grav);
132 
133  if (third_bodies_active) {
134  third_body_adjustments(real_time,
135  veh_struc_frame);
136  }
137 
138  // It appears that this would be better handled by polymorphism - that a call
139  // to, say, update_surface, would branch to the appropriate method, which
140  // should be in RadiationSurface, not RadiationPressure. However, the
141  // methods called here are more managerial that operational. The operational
142  // methods are in the appropriate Surface class, and are called by these
143  // managerial methods. They do belong in RadiationPressure.
144  if (surface_ptr == nullptr) {
145  update_default_surface();
146  }
147  else {
148  update_facet_surface();
149  }
150  return;
151 }

153 /**
154  * Used to modify the flux for shadowing.
155  * \param[in] real_time The current time, e.g. TAI or UT1.\n Units: s
156  * \param[in] veh_struc_frame Structure frame of vehicle
157  */
158 void
159 RadiationPressure::third_body_adjustments (
160  double real_time,
161  RefFrame & veh_struc_frame)
162 {
163  illum_factor = 1.0;
164 // multi_shadows = false;
165  for (unsigned int ii_body = 0; ii_body < num_third_bodies; ++ii_body) {
166  if (third_bodies[ii_body]->active) {
167  illum_factor *= third_bodies[ii_body]->process_third_body( real_time,
168  veh_struc_frame);
169  }
170  }
171  source.flux_mag = source.flux_mag * illum_factor;
172  Vector3::scale(illum_factor, source.flux_inertial);
173  Vector3::scale(illum_factor, source.flux_struc);
174 
175  return;
176 }

178 /**
179  * Adds a third body to the vector storage of third bodies
180  * \param[in] third_body_ptr pointer to ThirdBody
181  */
182 void
183 RadiationPressure::add_third_body (
184  RadiationThirdBody * third_body_ptr)
185 {
186  if (third_body_ptr == nullptr) {
187  MessageHandler::fail (
188  __FILE__, __LINE__, RadiationMessages::invalid_setup_error, "\n"
189  "Attempted to push a NULL object onto the ThirdBody vector in "
190  "the Radiation Pressure model.\n");
191  return;
192  }
193  // third_body_ptr is not NULL
194  // If third body is not named, then fail at initialization, and flag a
195  // warning that the adding has failed (but continue with the simulation)
196  // otherwise.
197  if (third_body_ptr->name.empty()) {
198  if (initialized) {
199  MessageHandler::error (
200  __FILE__, __LINE__, RadiationMessages::incomplete_setup_error, "\n"
201  "Attempting to add a ThirdBody with no defined name.\n"
202  "This cannot be done, since the name is needed to identify the "
203  "reference frame.\n"
204  "Addition of ThirdBody has failed. Simulation proceeding without it."
205  "\n");
206  return;
207  }
208  else {
209  MessageHandler::fail (
210  __FILE__, __LINE__, RadiationMessages::incomplete_setup_error, "\n"
211  "Attempting to add a ThirdBody with no defined name.\n"
212  "This cannot be done, since the name is needed to identify the "
213  "reference frame.\n");
214  }
215  }
216  // Check that the Third Body has not already been added. The
217  // RadiaitonThirdBody points to the RadiationSource, which provides
218  // information about the local radiation environment. Therefore, the
219  // RadiationSource is model specific, and therefore the RadiationThirdBody
220  // must also be model-specific. They cannot be shared across models.
221  if (third_body_ptr->get_added_to_model() ) {
222  MessageHandler::fail (
223  __FILE__, __LINE__, RadiationMessages::invalid_setup_error, "\n"
224  "Attempting to add a ThirdBody to more than one Radiation Pressure "
225  "model.\n"
226  "Offending object is called(%s).\n"
227  "ThirdBody objects can only be added to one Radiation Pressure model.\n"
228  "The same entity (vehicle, planet) can be represented as a ThirdBody\n"
229  "in multiple RadiationPressure models, but each representation must\n"
230  "be a separate instance of a ThirdBody.\n"
231  "E.g., there can be two ThirdBody objects called(%s), each linked to the\n"
232  "appropriate object (Planet or DynBody) of the same name, so that two \n"
233  "vehicles can each have their own RadiationPressure model with a \n"
234  "representation of the effects that(%s) has on their local radiation "
235  "field.\n",
236  third_body_ptr->name.c_str(), third_body_ptr->name.c_str(), third_body_ptr->name.c_str());
237  }
238 
239  // Don't try adding two ThirdBody objects with the same name to one model.
240  // This is usually an error of duplicity in setting up the model, so
241  // duplicates are just ignored.
242  for (unsigned int ii_t_b = 0; ii_t_b < num_third_bodies; ii_t_b++) {
243  if (third_body_ptr->name == third_bodies[ii_t_b]->name.c_str()) {
244  MessageHandler::error (
245  __FILE__, __LINE__, RadiationMessages::invalid_setup_error,"\n"
246  "In the Radiation Pressure Model, two RadiationThirdBody "
247  "objects \n were found with identical names (%s).\n "
248  "Omitting one.\n", third_body_ptr->name.c_str());
249  return;
250  }
251  }
252 
253  third_bodies.push_back(third_body_ptr);
254  third_body_ptr->set_added_to_model(true);
255 
256  // If the model has already been initialized, then initialize the third
257  // body (else, the third bodies will all be initialized with the
258  // model, so no action needed).
259 
260 
261  if (initialized) {
262  third_body_ptr->initialize (dyn_manager_ptr);
263  }
264 
265  // If any ThirdBody is active, the model should be looking for it/them.
266  if (third_body_ptr->active) {
267  third_bodies_active = true;
268  }
269 
270  num_third_bodies = third_bodies.size();
271  return;
272 }

275 /**

279 void
280 RadiationPressure::set_third_body_active (
281  const char * third_body_name)
282 {
283  int body_index = find_third_body( third_body_name );
284  if (body_index >= 0) {
285  if (third_bodies[body_index]->active) {
286  MessageHandler::warn (
287  __FILE__, __LINE__, RadiationMessages::invalid_function_call, "\n"
288  "Call made to set the RadiationThirdBody(%s) active, but it is "
289  "already active.\n",third_body_name);
290  }
291  else {
292  third_bodies_active = true;
293  third_bodies[body_index]->active = true;
294  if (third_bodies[body_index]->local_frame_ptr != nullptr) {
295  dyn_manager_ptr->subscribe_to_frame (
296  *third_bodies[body_index]->local_frame_ptr);
297  }
298  }
299  }
300  else {
301  // Continuation from find_third_body error message
302  MessageHandler::error (
303  __FILE__, __LINE__, RadiationMessages::invalid_function_call, "\n"
304  "Failure to find third body by name means it cannot be activated.");
305  }
306 
307  return;
308 }

319 {

321  if (body_index >= 0) {
322  if (!third_bodies[body_index]->active) {

333  }
334  third_bodies_active = false;
335  for (unsigned int ii = 0; ii < num_third_bodies; ii++ ) {
336  if (third_bodies[ii]->active) {

341  }
342  else {
343  // Continuation from find_third_body error message
344  MessageHandler::error (

346  "Failure to find third body by name means it cannot be deactivated.");

357 int
358 RadiationPressure::find_third_body (
359  const char * third_body_name)
360 {
361  for (unsigned int ii_body = 0; ii_body < num_third_bodies; ++ii_body) {
362  if (std::strcmp (third_bodies[ii_body]->name.c_str(), third_body_name) == 0) {
363  return ii_body;
364  }
365  }
366  MessageHandler::error (

370 
371  return -1;
