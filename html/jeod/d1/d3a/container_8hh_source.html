316  /**
317  * Perform a checkpoint-restart action that will, in part, restore the object
318  * to its state at the time of the checkpoint.
319  *
320  * In the case of a JeodContainer, the actions are "clear", "insert", and
321  * "cleanup". The checkpoint writer automatically creates an initial "clear"
322  * entry as the first entry in the checkpoint file for a JeodCheckpointable
323  * object and a "cleanup" entry as the final entry. An "insert" entry is
324  * created for each element in the container's contents.
325  */
326  int perform_restore_action (
327  const std::string & action_name,
328  const std::string & action_value) override
329  {
330  int return_code;
331 
332  // "clear" action response is to clear the contents of the object.
333  if (action_name.compare("clear") == 0) {
334  this->clear();
335  return_code = 0;
336  }
337 
338  // "insert" action response is to insert the value in the contents.
339  else if (action_name.compare("insert") == 0) {
340  perform_insert_action (action_value);
341  return_code = 0;
342  }
343 
344  // "cleanup" action response is to remove detritus created during restart.
345  else if (action_name.compare("cleanup") == 0) {
346  perform_cleanup_action (action_value);
347  return_code = 0;
348  }
349 
350  // All other actions are illegal and should never happen.
351  else {
352  return_code = -1;
353  }
354 
355  return return_code;
356  }
