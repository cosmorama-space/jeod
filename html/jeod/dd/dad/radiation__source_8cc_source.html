98  Purpose:
99  (calculates the flux vector from the vehicle's position.)
100 *****************************************************************************/
101 void
102 RadiationSource::calculate_flux (
103  RefFrame& veh_struc_frame,
104  const double center_gravity[3])
105 {
106  if (luminosity < 1.0E-6) {
107  flux_mag = 0.0;
108  return;
109  }
110 
111  // Calculate the position of the vehicle with respect to Primary Source.
112  // Start by getting the position of the origin of the vehicle structural
113  // frame.
114  veh_struc_frame.compute_position_from (*inertial_frame_ptr,
115  source_to_struc_origin);
116 
117  // center_gravity is in structural. Need it in inertial to complete the
118  // position vector (center_gravity is more closely set to the vehicle than the
119  // origin of the structural frame). Since the parent-frame of any root
120  // vehicle is an inertial-frame, the vehicle's T_parent_this will convert
121  // structural to inertial.
122  Vector3::transform_transpose (veh_struc_frame.state.rot.T_parent_this,
123  center_gravity,
124  inertial_cg);
125 
126  // Add the vector to structural origin to the vector from structural origin to
127  // vehicle center.
128  Vector3::sum (source_to_struc_origin,
129  inertial_cg,
130  source_to_cg);
131 
132  // Now normalize this vector. Since we need the magnitude later, do this in
133  // two steps, rather than use the available "normalize" function.
134 
135  d_source_to_cg = Vector3::vmag (source_to_cg);
136 
137  Vector3::scale (source_to_cg,
138  1 / d_source_to_cg,
139  flux_hat);
140 
141 
142 
143  flux_mag = luminosity / (d_source_to_cg * d_source_to_cg * 4 * M_PI);
144 
145  Vector3::scale (flux_hat,
146  flux_mag,
147  flux_inertial);
148 
149  // Transform this vector back into structural reference and normalize.
150  Vector3::transform (veh_struc_frame.state.rot.T_parent_this,
151  flux_inertial,
152  flux_struc);
153 
154  Vector3::normalize (flux_struc,
155  flux_struc_hat);
156  return;
157 }
