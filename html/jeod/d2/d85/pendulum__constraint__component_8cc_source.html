172 bool
173 PendulumConstraintComponent::check_for_reset ()
174 {
175  // Need to reset the integrators if this has been forced from the outside.
176  if (new_mode == PendulumConstraintMode::Reset)
177  {
178  return true;
179  }
180 
181  // Need to reset the integrators if the pendulum's integrated state is the
182  // cartesian state but it now would be better to integrate angular state.
183  else if (((mode == PendulumConstraintMode::Vertical) ||
184  (mode == PendulumConstraintMode::Inverted)) &&
185  (cos_sq_theta < cos_sq_theta_angular))
186  {
187  new_mode = PendulumConstraintMode::Horizontal;
188  return true;
189  }
190 
191  // Need to reset the integrators if the pendulum's integrated state is the
192  // angular state but it now would be better to integrate cartesian state.
193  else if ((mode == PendulumConstraintMode::Horizontal) &&
194  (cos_sq_theta > cos_sq_theta_cartesian))
195  {
196  if (bob_position[2] > 0.0)
197  {
198  new_mode = PendulumConstraintMode::Vertical;
199  }
200  else
201  {
202  new_mode = PendulumConstraintMode::Inverted;
203  }
204  return true;
205  }
206 
207  else
208  {
209  new_mode = mode;
210  return false;
211  }
212 }

215 void
216 PendulumConstraintComponent::transition_to_horizontal ()
217 {
218  double rsq = bob_position[0]*bob_position[0] + bob_position[1]*bob_position[1];
219  double r = std::sqrt(rsq);
220  sin_theta = r / pendulum_length;
221  cos_theta = bob_position[2] / pendulum_length;
222  sin_phi = bob_position[1] / r;
223  cos_phi = bob_position[0] / r;
224  theta = std::acos(cos_theta);
225  phi = std::atan2(sin_phi, cos_phi);
226  construct_constraint_to_spherical_transform ();
227  theta_dot = Vector3::dot(
228  bob_velocity, constraint_to_spherical_transform[1])
229  / pendulum_length;
230  phi_dot = Vector3::dot(
231  bob_velocity, constraint_to_spherical_transform[2])
232  / (pendulum_length*sin_theta);
233  compute_rho_hat_dot ();
234 }

395 void
396 PendulumConstraintComponent::compute_bob_state_cartesian ()
397 {
398  // The above only integrated the x & y position and velocity.
399  // Compute the z position and velocity.
400  double rsq = bob_position[0]*bob_position[0] + bob_position[1]*bob_position[1];
401  double lsq = pendulum_length*pendulum_length;
402  double zsq = lsq - rsq;
403 
404  bob_position[2] = std::sqrt(zsq);
405  bob_velocity[2] =
406  -(bob_position[0]*bob_velocity[0] + bob_position[1]*bob_velocity[1])
407  / bob_position[2];
408 
409  if (mode == PendulumConstraintMode::Inverted)
410  {
411  bob_position[2] = -bob_position[2];
412  bob_velocity[2] = -bob_velocity[2];
413  }
414 
415  cos_theta = bob_position[2] / pendulum_length;
416  cos_sq_theta = cos_theta * cos_theta;
417 
418  // Set the constraint direction for the next time around.
419  double bob_direction[3];
420  Vector3::scale (bob_position, 1.0/pendulum_length, bob_direction);
421  set_constraint_direction (bob_direction);
422 }

425 void
426 PendulumConstraintComponent::compute_constraint_response (
427  const VehicleProperties&,
428  const VehicleNonGravState& non_grav_state)
429 {
430  constrained_mass->compute_non_grav_accel (non_grav_state);
431 
432  double tension_accel = constraint_force / constrained_mass->mass;
433  double damping_factor = pendulum_model->compute_damping_factor(
434  Vector3::vmag(constrained_mass->non_grav_accel_root));
435  // Or std::abs(tension_accel)
436 
437  /*
438  * Note: Tension accel could be calculated using the same expression
439  * used in setup_constraint(), but there's no point. The solution
440  * to the constraints system of equations for this constrained object
441  * is the tension force.
442  */
443 
444  Vector3::transform (
445  constraint_frame->T_root_constraint,
446  non_grav_state.omega_struct,
447  omega_constraint_frame);
448  Vector3::transform (
449  constraint_frame->T_root_constraint,
450  non_grav_state.omega_dot_struct,
451  omega_dot_constraint_frame);
452 
453  if (mode == PendulumConstraintMode::Horizontal)
454  {
455  double omega_rho = Vector3::dot(constraint_to_spherical_transform[0],
456  omega_constraint_frame);
457  double accel_theta = Vector3::dot(constraint_to_spherical_transform[1],
458  constrained_mass->non_grav_accel_constraint_frame);
459  double omega_theta = Vector3::dot(constraint_to_spherical_transform[1],
460  omega_constraint_frame);
461  double omega_dot_theta = Vector3::dot(constraint_to_spherical_transform[1],
462  omega_dot_constraint_frame);
463  double accel_phi = Vector3::dot(constraint_to_spherical_transform[2],
464  constrained_mass->non_grav_accel_constraint_frame);
465  double omega_phi = Vector3::dot(constraint_to_spherical_transform[2],
466  omega_constraint_frame);
467  double omega_dot_phi = Vector3::dot(constraint_to_spherical_transform[2],
468  omega_dot_constraint_frame);
469 
470  theta_dot_dot = -(
471  damping_factor*theta_dot
472  + accel_theta/pendulum_length
473  + omega_rho*(omega_theta - 2.0*phi_dot*sin_theta)
474  + omega_dot_phi
475  - phi_dot*phi_dot*sin_theta*cos_theta);
476  phi_dot_dot = -(
477  damping_factor*phi_dot
478  + ( accel_phi/pendulum_length
479  + omega_rho*(omega_phi + 2.0*theta_dot)
480  - omega_dot_theta
481  + 2.0*theta_dot*phi_dot*cos_theta
482  ) / sin_theta);
483  }
484  else
485  {
486  double tension_accel_div_length = tension_accel / pendulum_length;
487  double omega_dot_cross_pos[3];
488  double omega_cross_omega_cross_pos[3];
489  double* non_grav_accel_constraint_frame =
490  constrained_mass->non_grav_accel_constraint_frame;
491 
492  Vector3::cross(omega_dot_constraint_frame, bob_position, omega_dot_cross_pos);
493  Vector3::cross(
494  omega_constraint_frame, omega_cross_pos, omega_cross_omega_cross_pos);
495 
496  for (int ii = 0; ii < 2; ++ii)
497  {
498  // Calculation is done in two steps for numerical reasons.
499  // The first part will typically be smaller than the second.
500  bob_accel[ii] =
501  -( damping_factor * bob_velocity[ii] // Drag
502  + omega_dot_cross_pos[ii] // Euler
503  + 2.0*omega_cross_vel[ii] // Coriolis
504  + omega_cross_omega_cross_pos[ii]) // Centrifugal
505  -( tension_accel_div_length * bob_position[ii] // Tension
506  + non_grav_accel_constraint_frame[ii]); // Inertial
507  }
508  }
509 
510  double damping_torque_constraint_frame[3];
511  double damping_torque_root[3];
512  Vector3::cross (bob_position, bob_velocity, damping_torque_constraint_frame);
513  Vector3::scale (
514  constrained_mass->mass * damping_factor,
515  damping_torque_constraint_frame);
516  Vector3::transform_transpose (
517  constraint_frame->T_root_constraint,
518  damping_torque_constraint_frame,
519  damping_torque_root);
520  damping_wrench.set_torque(damping_torque_root);
521 }
