102 /**
103  * Begin initialization of a LvlhDerivedState.
104  * The initialize method for all subclasses of DerivedState *must* pass
105  * the initialize call to their immediate parent class, which in turn
106  * must do the same, eventually invoking this method.
107  * \param[in,out] subject_body Subject body
108  * \param[in,out] dyn_manager Dynamics manager
109  */
110 void
111 NedDerivedState::initialize (
112  DynBody & subject_body,
113  DynManager & dyn_manager)
114 {
115  // Initialize as a DerivedState.
116  DerivedState::initialize (subject_body, dyn_manager);
117 
118  // Find the planet.
119  // Note that find_planet doesn't return if the named planet is not found.
120  planet = find_planet (dyn_manager, reference_name, "reference_name");
121 
122  // Choose the planet fixed frame to be used
123  if (use_alt_pfix) {
124  pfix_ptr = &(planet->alt_pfix);
125  } else {
126  pfix_ptr = &(planet->pfix);
127  }
128  pfix_ptr->subscribe();
129 
130  // Name the frame as <vehicle>.<planet>.ned
131  ned_state.ned_frame.set_name (subject_body.name.c_str(), reference_name, "ned");
132 
133  // Connect the frame to the planet's planet-fixed frame
134  // and subscribe to that frame.
135  pfix_ptr->add_child (ned_state.ned_frame);
136 
137  // If requested, register the frame with the dynamics manager.
138  if (register_frame) {
139  dyn_manager.add_ref_frame (ned_state.ned_frame);
140  }
141 
142  // Initialize the NED state.
143  ned_state.initialize (planet);
144 }
