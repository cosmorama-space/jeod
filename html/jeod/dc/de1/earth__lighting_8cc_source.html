266 FUNCTION: circle_intersect
267 PURPOSE:
268  (This routine computes the area of intersection of two overlapping
269  circles.)
270 
271 
272 ASSUMPTIONS AND LIMITATIONS:
273  ((EPSILON comparison in if test help protect against a numerical
274  singularity for d_centers = 0.0 ))
275 
276 ******************************************************************************/
277 int
278 EarthLighting::circle_intersect (
279  double r_bottom,
280  double r_top,
281  double d_centers,
282  double* area)
283 {
284 
285  double d_c2;
286  double r_t2;
287  double r_b2;
288  double diff_r2;
289  double top_ang;
290  double cos_top_ang;
291  double top_area;
292  double bottom_ang;
293  double cos_bottom_ang;
294  double bottom_area;
295 
296  /***************************************************/
297  /* Quick check to see if circles are intersecting. */
298  /***************************************************/
299  if (d_centers > r_bottom + r_top) {
300 
301  /* Compute the area of intersection. */
302  *area = 0.0;
303 
304  /* Return negative intersection flag. */
305  return(0);
306 
307  }
308 
309  /*************************************************************/
310  /* Check to see if one circle completely contains the other. */
311  /*************************************************************/
312  if (r_bottom > r_top) {
313 
314  /* Check to see if top circle is completely inside bottom circle. */
315  if (d_centers < (r_bottom - r_top) + EPSILON) {
316 
317  /* Area of intersection is entire top circle. */
318  *area = M_PI * r_top * r_top;
319 
320  /* Return positive intersection flag. */
321  return(1);
322 
323  }
324 
325  }
326  else {
327 
328  /* Check to see if top circle completely covers bottom circle. */
329  if (d_centers < (r_top - r_bottom) + EPSILON) {
330 
331  /* Area of intersection is entire bottom circle. */
332  *area = M_PI * r_bottom * r_bottom;
333 
334  /* Return positive intersection flag. */
335  return(1);
336 
337  }
338 
339  }
340 
341  /*********************************************************************/
342  /* Compute the area of intersection of the two overlappping circles. */
343  /*********************************************************************/
344 
345  /* Compute working variables. */
346  d_c2 = d_centers * d_centers;
347  r_t2 = r_top * r_top;
348  r_b2 = r_bottom * r_bottom;
349  diff_r2 = r_b2 - r_t2;
350 
351  /* Compute the bottom intersection angle (law of cosines). */
352  cos_bottom_ang = (d_c2 + diff_r2) / (2.0 * d_centers * r_bottom);
353  bottom_ang = acos (cos_bottom_ang);
354 
355  /* Compute the top intersection angles (law of cosines). */
356  cos_top_ang = (d_c2 - diff_r2) / (2.0 * d_centers * r_top);
357  top_ang = acos (cos_top_ang);
358 
359  /* Compute the area of intersection. */
360  top_area = r_t2 * (top_ang - (sin (top_ang) * cos_top_ang));
361  bottom_area = r_b2 * (bottom_ang - (sin (bottom_ang) * cos_bottom_ang));
362  *area = top_area + bottom_area;
363 
364  /* Return a positive for intersection. */
365  return(1);
366 
367 }

374 void
375 EarthLighting::calc_lighting (
376  const double pos_veh[3])
377 {
378 
379  if (active == false) {
380  return;
381  }
382 
383  int iinc;
384 
385  double eclipse_area;
386 
387  double cross_prod[3];
388  double cross_mag;
389  double cos_obs_ang;
390  double sin_obs_ang;
391 
392  // Get the necessary relative positions
393  sun_frame->compute_position_from (*earth_frame, pos_sun);
394  moon_frame->compute_position_from (*earth_frame, pos_moon);
395 
396  /* Compute the relative positions of the celestial bodies. */
397  for (iinc = 0; iinc < 3; iinc++) {
398  moon_body.position[ iinc ] = pos_moon[ iinc ] - pos_veh[ iinc ];
399  sun_body.position[ iinc ] = pos_sun[ iinc ] - pos_veh[ iinc ];
400  earth_body.position[ iinc ] = -pos_veh[ iinc ];
401  }
402 
403  /* Compute the distance to the celestial bodies. */
404  moon_body.distance = Vector3::vmag (moon_body.position);
405  sun_body.distance = Vector3::vmag (sun_body.position);
406  earth_body.distance = Vector3::vmag (earth_body.position);
407 
408  /* Compute the apparent half angles of the celestial bodies. */
409  moon_body.half_angle = asin (moon_body.radius / moon_body.distance);
410  sun_body.half_angle = asin (sun_body.radius / sun_body.distance);
411  if (earth_body.distance >= earth_body.radius) {
412  earth_body.half_angle = asin (earth_body.radius / earth_body.distance);
413  }
414  else {
415  earth_body.half_angle = M_PI_2;
416  }
417 
418  /**********************************************************************/
419  /* Compute the observation angle between the earth and light sources. */
420  /**********************************************************************/
421  /* Moon */
422 // cos_obs_ang = V_DOT( moon_body.position, earth_body.position )
423 // / ( moon_body.distance* earth_body.distance );
424  cos_obs_ang = Vector3::dot (moon_body.position, earth_body.position) /
425  (moon_body.distance * earth_body.distance);
426  // V_CROSS( cross_prod, moon_body.position, earth_body.position );
427  Vector3::cross (moon_body.position, earth_body.position, cross_prod);
428  cross_mag = Vector3::vmag (cross_prod);
429  sin_obs_ang = cross_mag / (moon_body.distance * earth_body.distance);
430  moon_earth.obs_angle = atan2 (sin_obs_ang, cos_obs_ang);
431 
432  /* Sun */
433 // cos_obs_ang = V_DOT( sun_body.position, earth_body.position )
434 // / ( sun_body.distance* earth_body.distance );
435  cos_obs_ang = Vector3::dot (sun_body.position, earth_body.position) /
436  (sun_body.distance * earth_body.distance);
437 // V_CROSS( cross_prod, sun_body.position, earth_body.position );
438  Vector3::cross (sun_body.position, earth_body.position, cross_prod);
439  cross_mag = Vector3::vmag (cross_prod);
440  sin_obs_ang = cross_mag / (sun_body.distance * earth_body.distance);
441  sun_earth.obs_angle = atan2 (sin_obs_ang, cos_obs_ang);
442 
443  /* Determine if earth occludes the sun. */
444  /* Note: these represent arc-lengths on a unit sphere. */
445  circle_intersect (sun_body.half_angle,
446  earth_body.half_angle,
447  sun_earth.obs_angle,
448  &eclipse_area);
449 
450  /* Compute the sun lighting from the eclipse area. */
451  sun_earth.occlusion =
452  eclipse_area / (sun_body.half_angle * sun_body.half_angle * M_PI);
453  sun_earth.visible = 1.0 - sun_earth.occlusion;
454  sun_earth.lighting = sun_earth.phase * sun_earth.visible;
455 
456  /* Determine if earth occludes the moon. */
457  /* Note: these represent arc-lengths on a unit sphere. */
458  circle_intersect (moon_body.half_angle,
459  earth_body.half_angle,
460  moon_earth.obs_angle,
461  &eclipse_area);
462 
463  /* Compute the moon lighting from the eclipse area. */
464  moon_earth.occlusion =
465  eclipse_area / (moon_body.half_angle * moon_body.half_angle * M_PI);
466  moon_earth.visible = 1.0 - moon_earth.occlusion;
467 
468  /* Apply further scaling by apparent lunar phase. */
469  moon_earth.lighting = moon_earth.phase * moon_earth.visible;
470 
471  /* Crude approximation for Earth albedo. */
472  earth_albedo.lighting = fabs (sun_earth.obs_angle / M_PI);
473  earth_albedo.lighting *= sun_earth.lighting;
474 
475  return;
476
