91 /**
92  * Overloaded functions that create a ContactPair and pass the address
93  * of it to the Contact class for addition to the list of pairs.
94  * This function is called when a subject and target are known.
95  * @return ContactPair that was created
96  * \param[in,out] target target ContactFacet

98  */
99 ContactPair *
100 PointContactFacet::create_pair (
101  ContactFacet * target,
102  Contact * contact)
103 {
104  PointContactFacet * temp_ptr;
105 
106  // attempt to cast the target to a PointContactFacet
107  temp_ptr = dynamic_cast<PointContactFacet*> (target);
108  if (temp_ptr == nullptr) {
109  MessageHandler::warn (
110  __FILE__, __LINE__, ContactMessages::initialization_warns,
111  "The target ContactFacet passed into PointContactFacet::create_pair was "
112  "not of type PointContactFacet as required.");
113 
114  return nullptr;
115  }
116 
117  PointContactPair * contact_pair;
118  // Allocate a ContactPair instance
119  contact_pair = JEOD_ALLOC_CLASS_OBJECT (PointContactPair, ());
120 
121  contact_pair->interaction = contact->find_interaction(surface_type, target->surface_type);
122  if (contact_pair->interaction == nullptr) {
123  return nullptr;
124  }
125  // set the subject and target of the new pair
126  contact_pair->initialize_pair(this, temp_ptr);
127 
128  contact_pair->interaction_distance = (max_dimension + temp_ptr->max_dimension) * contact->contact_limit_factor;
129 
130  return contact_pair;
131 }

161 /**
162  * Calculate the torque generated on the vehicle by the facet. Assumes
163  * that the force is in the vehicle structural frame, but that close_point is not.
164  * \param[in] tmp_force force from one contact interaction.\n Units: N
165  */
166 void PointContactFacet::calculate_torque (
167  double *tmp_force)
168 {
169  double cm[3];
170  double vec[3];
171  double tmp_torque[3];
172 
173  // find the vector from vehicle point to center of mass
174  Vector3::diff(vehicle_point->mass_point->position,
175  vehicle_body->mass.composite_properties.position,
176  cm);
177 
178  // rotate facet contact point to vehicle structure frame
179  Vector3::transform_transpose (vehicle_point->state.rot.T_parent_this,
180  contact_point, vec);
181  Vector3::transform (vehicle_body->structure.state.rot.T_parent_this,
182  vec, tmp_torque);
183  // find vector from cm to contact point
184  Vector3::sum(tmp_torque, cm, cm);
185  // cross cm vector with the rotated force to get the torque
186  Vector3::cross (cm, tmp_force, tmp_torque);
187  Vector3::sum (torque, tmp_torque, torque);
188 
189  return;
190 }
