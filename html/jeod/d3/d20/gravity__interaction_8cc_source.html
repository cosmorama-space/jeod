190 /**
191  * Sort the GravityControls in the grav_controls list
192  * in increasing acceleration magnitude order.
193  */
194 void
195 GravityInteraction::sort_controls (
196  void)
197 {
198 
199  // Compute the magnitude of each gravity acceleration vector.
200  for (unsigned int ii = 0; ii < grav_controls.size(); ++ii) {
201  GravityControls & control_ii = *grav_controls[ii];
202  if (control_ii.active) {
203  control_ii.grav_accel_magsq = Vector3::vmagsq(control_ii.grav_accel);
204  }
205  else {
206  control_ii.grav_accel_magsq = 0.0;
207  }
208  }
209 
210  // Sort the controls in increasing acceleration order.
211  std::sort (grav_controls.begin(), grav_controls.end(),
212  GravityControls::accel_mag_less_ptr);

214  return;
