112 // Destruct a DynBody.
113 DynBody::~DynBody ()
114 {
115  // Detach this body from its parent.
116  if (dyn_parent != nullptr) {
117  detach (*dyn_parent);
118  }
119 
120  // Detach each child DynBody from this body.
121  while (! dyn_children.empty()) {
122  DynBody* child = dyn_children.back();
123  child->detach (*this);
124  }
125  // Detach each sub MassBody from this body.
126  while (! mass_children.empty()) {
127  MassBody* sub = mass_children.back();
128  remove_mass_body(*sub);
129  }
130 
131  // Remove the body reference frames from the ref frame tree.
132  core_body.remove_from_parent ();
133  composite_body.remove_from_parent ();
134  structure.remove_from_parent ();
135  if( dyn_manager )
136  {
137  dyn_manager->remove_ref_frame( core_body );
138  dyn_manager->remove_ref_frame( composite_body );
139  dyn_manager->remove_ref_frame( structure );
140  }
141 
142  // Remove the registrations of the checkpointable objects.
143  JEOD_DEREGISTER_CHECKPOINTABLE (this, trans_integrator);
144  JEOD_DEREGISTER_CHECKPOINTABLE (this, rot_integrator);
145 
146  // Delete the allocated vehicle points, in reverse order.
147  while (! vehicle_points.empty()) {
148  auto* point = vehicle_points.back();
149  if( dyn_manager )
150  {
151  dyn_manager->remove_ref_frame( *point );
152  }
153  if (JEOD_IS_ALLOCATED (point)) {
154  JEOD_DELETE_OBJECT (point);
155  }
156  vehicle_points.pop_back();
157  }
