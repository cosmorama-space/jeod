120 /**
121  * Given a value of seconds, propagate to other reps.
122  *
123  * \par Assumptions and Limitations
124  * - 86400 seconds = 1 day
125  * \param[in] new_seconds new value for seconds\n Units: s
126  */
127 void
128 TimeGPS::set_time_by_seconds (
129  const double new_seconds)
130 
131 {
132  TimeStandard::set_time_by_seconds (new_seconds);
133  // Split time into day number and seconds of day.
134  int gps_time_int = static_cast<int> (days);
135  seconds_of_day = (days - gps_time_int) * 86400.0;
136 
137  // GPS time rolls over every 1024 weeks, or every 7168 days.
138  rollover_count = gps_time_int / 7168;
139  // In newer systems, GPS time rolls over every 8192 weeks, or 57344 days.
140  rollover_count_13_bit = gps_time_int / 57344;
141  week_13_bit = (gps_time_int-57344*rollover_count_13_bit)/7;
142  gps_time_int -= rollover_count * 7168;
143  week = gps_time_int / 7; // weeks since rollover
144 
145  day_of_week =
146  gps_time_int - week * 7; /* number of full days in this
147  (partial) week */
148  seconds_of_week =
149  day_of_week * 86400 + seconds_of_day; // convert to seconds
150 
151  return;
152 
153 }
