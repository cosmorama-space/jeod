315  /**
316  * Construct a name from the given input, as a string.
317  * The input must not be the empty string or the null pointer.
318  * @tparam Arg Type of the argument to construct_name_string.
319  * @param arg Argument to construct_name_string.
320  * @return std::string that is conceptually equal to (==) arg.
321  */
322  template <typename Arg>
323  static std::string construct_name_string (Arg&& arg)
324  {
325  std::string result { std::forward<Arg>(arg) };
326  validate_name (__FILE__, __LINE__, result.c_str(), "Label", "arg");
327  return result;
328  }
329 
330  /**
331  * Construct a name as a dot-conjoined string of the given inputs.
332  * Each input must not be the empty string or the null pointer.
333  * @tparam First Type of the first argument to construct_name_string.
334  * @tparam Rest Types of the remaining arguments to construct_name_string.
335  * @param first First argument to construct_name_string.
336  * @param rest Remaining arguments to construct_name_string.
337  * @return The given inputs as a dot-conjoined string.
338  */
339  template <typename First, typename... Rest>
340  static std::string construct_name_string (First&& first, Rest&&... rest)
341  {
342  std::string result { std::forward<First>(first) };
343  validate_name (__FILE__, __LINE__, result.c_str(), "Label", "arg");
344  std::vector<std::string> rest_vec { { std::forward<Rest>(rest)... } };
345  for (auto& elem : rest_vec)
346  {
347  result += '.';
348  validate_name (__FILE__, __LINE__, elem.c_str(), "Elem", "elem");
349  result += elem;
350  }
351  return result;
352  }
