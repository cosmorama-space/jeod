298  * Apply the corrector.
299  */
300 template<>
301 inline void
302 GaussJacksonIntegratorBase <
303 GaussJacksonTwoState,
304 er7_utils::SecondOrderODEIntegrator>::correct (
305  double dt,
306  const double* ER7_UTILS_RESTRICT acc,
307  GaussJacksonTwoState & state)
308 {
309  double dtsq = dt*dt;
310  double* ER7_UTILS_RESTRICT first_csum = corrector_sum.first;
311  double* ER7_UTILS_RESTRICT second_csum = corrector_sum.second;
312  double* ER7_UTILS_RESTRICT first_dinv = delinv.first;
313  double* ER7_UTILS_RESTRICT second_dinv = delinv.second;
314  double* ER7_UTILS_RESTRICT velocity = state.first;
315  double* ER7_UTILS_RESTRICT position = state.second;
316  double vfact = velocity_corrector;
317  double pfact = position_corrector;
318 
319  for (unsigned int ii = 0; ii < size; ++ii) {
320 
321  double temp = first_csum[ii] + vfact*acc[ii];
322  velocity[ii] = dt * (first_dinv[ii] + temp);
323 
324  temp = second_csum[ii] + pfact*acc[ii];
325  position[ii] = dtsq * (second_dinv[ii] + temp);
326  }
327 }
