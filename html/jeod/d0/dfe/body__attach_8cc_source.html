118 /**
119  * A derived class presumably has performed the attachment, which may
120  * not have worked, and forwarded the apply call to this method.
121  * This method acts on the status from that child class attachment.
122  * \param[in,out] dyn_manager Jeod manager
123  */
124 void
125 BodyAttach::apply (
126  DynManager & dyn_manager)
127 {
128  std::string subject_name("Unknown");
129  std::string parent_name("Unknown");
130  if (dyn_parent != nullptr) {
131  parent_name = dyn_parent->name.get_name();
132  } else {
133  if (mass_parent != nullptr) {
134  parent_name = mass_parent->name.get_name();
135  } else if (ref_parent != nullptr) {
136  parent_name = ref_parent->get_name();
137  }
138  }
139 
140  if (dyn_subject != nullptr) {
141  subject_name = dyn_subject->name.get_name();
142  } else {
143  if (mass_subject != nullptr) {
144  subject_name = mass_subject->name.get_name();
145  }
146  }
147 
148  // Attachment succeeded: Debug.
149  if (succeeded) {
150  MessageHandler::debug (
151  __FILE__, __LINE__, BodyActionMessages::trace,
152  "%s: %s attached to %s.",
153  action_identifier.c_str(), subject_name.c_str(), parent_name.c_str());
154  }
155 
156  // Attachment failed: Terminate the sim if terminate_on_error is set.
157  else if (terminate_on_error) {
158  MessageHandler::fail (
159  __FILE__, __LINE__, BodyActionMessages::fatal_error,
160  "'%s' failed to attach '%s' to '%s'. "
161  "The terminate_on_failure flag set and an attachment error occurred.\n"
162  "The attachment error described above is fatal per this setting.",
163  action_identifier.c_str(), subject_name.c_str(), parent_name.c_str());
164  }
165 
166  // Attachment failed, terminate_ not set: Tell the user about the problem.
167  else {
168  MessageHandler::error (
169  __FILE__, __LINE__, BodyActionMessages::not_performed,
170  "%s failed to attach %s to %s.",
171  action_identifier.c_str(), subject_name.c_str(), parent_name.c_str());
172  }
173 
174 
175  // Forward the action up the chain.
176  BodyAction::apply (dyn_manager);
177 
178  return;
179 }
