145 /**
146  * A Subscription object provides two approaches of marking something as
147  * being active or inactive: The activate and deactivate methods versus the
148  * subscribe and unsubscribe methods. The class also provides a mean for
149  * selecting only one of these two approaches as valid.
150  *
151  * This class uses the non-virtual interface design pattern. Derived classes
152  * should not override the non-virtual public interfaces. They should instead
153  * override the private set_active_state method.
154  */
155 class Subscription {

162  /**
163  * The Subscription::Mode enum specifies the mode in which a Subscription
164  * object is operating.
165  */
166  enum Mode {
167  Detect = 0, ///< First scheme used wins.
168  Subscribe = 1, ///< Activation is via subscribe/unsubscribe only.
169  Activate = 2, ///< Activation is via activate/deactivate only.
170  Freeform = 3 ///< Users can use either scheme; conflicts may arise.
171  };
